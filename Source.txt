.MODEL SMALL
.STACK 100H
.DATA
AS DB "ENTER HEX ALPHABET :$"
D DB 0AH,0DH, "VALUE IN DECIMAL IS: $"
P DB "ENTER ANOTHER VALUE ? (Y/N):$"
.CODE
MAIN PROC
MOV AX,@DATA
MOV DS,AX
START:
MOV AH,2
MOV DL,10
INT 21H
LEA DX,AS
MOV AH,9
INT 21H
MOV AH,1
INT 21H
MOV BL,AL
MOV AH,2
LEA DX,D
MOV AH,9
INT 21H
MOV AH,2
MOV AL,BL
MOV DL,'1'
INT 21H
MOV DL,BL
SUB DL,17
INT 21H
ASK:
MOV DL,10
INT 21H
LEA DX,P
MOV AH,9
INT 21H

MOV AH,1
INT 21H

CMP AL,'Y'
JE START

EXIT:
MOV AH,4CH
INT 21H
MAIN ENDP
END MAIN
